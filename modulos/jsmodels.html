Name export individual

function mostraIdade(pessoa){
return "a idade dela ${pessoa.idade}
}

function mostraCidade(cidade){
return "a cidade dela ${pessoa.cidade}
}

export{
mostraIdade,
mostraCidade
}

Default export

Só pode haver um por arquivo, será o retorno padrão do seu arquivo


function mostraIdade(pessoa){
return "a idade dela ${pessoa.idade}
}

export default mostraIdade
